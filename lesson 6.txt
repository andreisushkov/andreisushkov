package lesson_6;

abstract class Animal {
    protected String name;
    public int run;



    public Animal(String name) {
        this.name = name;
    }

    public Animal() {

    }


    public void Info (){

    }
}



package lesson_6;

class Dog extends Animal {
    public double jump;
    public int sweem;





    public Dog(String name, int run,int sweem, double jump) {
        super(name);
        this.jump = jump;
        this.sweem = sweem;
        this.run = run;


    }

     void run () {
        if (run == 500) {
            System.out.println("run: true");
        } else {
            System.out.println("run: false");
        }
    }

    void jump () {
        if (jump == 0.5) {
            System.out.println("true");
        }else {
            System.out.println("false");
        }
    }

    void sweem() {
        if (sweem == 10) {
            System.out.println("sweem: true");
        } else {
            System.out.println("false");
        }
    }




    @Override
    public  void Info() {
        System.out.println("Dog_name: " + name + " run:"  + run + " jump:" + jump + " sweem " + sweem);

    }




}
